const ounceToGramRate = 28.3495;
const gramsResultDiv = document.getElementById("gramsResult");
const ouncesInput = document.getElementById("ouncesInput");

function convertToGrams() {
  const ounces = parseFloat(ouncesInput.value);
  if (!isNaN(ounces)) {
    const grams = (ounces * ounceToGramRate).toFixed(2);
    gramsResultDiv.textContent = `Grams: ${grams}`;
  } else {
    gramsResultDiv.textContent = "Grams: ";
  }
}
