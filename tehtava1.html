function isLeapYear(year) {
  year = parseInt(year, 10);

  if (isNaN(year)) {
    return false;
  }

  if (year % 4 === 0) {
    if (year % 100 !== 0 || year % 400 === 0) {
      return true;
    }
  }
  return false;
}

console.log("Is 2020 a leap year? " + isLeapYear(2020)); // Output: true
console.log("Is 2021 a leap year? " + isLeapYear(2021)); // Output: false
console.log("Is 1900 a leap year? " + isLeapYear(1900)); // Output: false
console.log("Is 2000 a leap year? " + isLeapYear(2000)); // Output: true
