Contains Number Checker

Code


var containsNumber = function(numbers, aNumber) {
  aNumber = parseInt(aNumber, 10);
  if (isNaN(aNumber) || !Array.isArray(numbers)) {
    return false;
  }
  for (var i = 0; i < numbers.length; i++) {
    if (numbers[i] === aNumber) {
      return true;
    }
  }
  return false;
};

var numbersToCheck = [6, 4, 2, 5, 9, 7, 5, 7, 2];
console.log("Does numbersToCheck contain 5? " + containsNumber(numbersToCheck, 5));   // Output: true
console.log("Does numbersToCheck contain 10? " + containsNumber(numbersToCheck, 10));  // Output: false
console.log("Does numbersToCheck contain NaN? " + containsNumber(numbersToCheck, NaN)); // Output: false
