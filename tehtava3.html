Convert to Minutes Format

Code


const convertToMinutesFormat = (hoursInHundredths) => {
  const hoursString = String(hoursInHundredths).replace(',', '.');
  const totalHundredths = parseFloat(hoursString);

  if (isNaN(totalHundredths)) {
    return "Invalid input";
  }

  const totalMinutes = Math.round(totalHundredths * 60);
  const hours = Math.floor(totalMinutes / 60);
  const minutes = totalMinutes % 60;

  const formattedHours = hours < 10 ? `0${hours}` : `${hours}`;
  const formattedMinutes = minutes < 10 ? `0${minutes}` : `${minutes}`;

  return `${hours}:${formattedMinutes}`;
};

console.log("2.5 hours to minutes format: " + convertToMinutesFormat(2.5));     // Output: 02:30
console.log("10.75 hours to minutes format: " + convertToMinutesFormat("10.75")); // Output: 10:45
console.log("0.25 hours to minutes format: " + convertToMinutesFormat("0,25"));   // Output: 00:15 (comma is replaced with a dot)
console.log("5 hours to minutes format: " + convertToMinutesFormat(5));         // Output: 05:00
console.log("Invalid input: " + convertToMinutesFormat("hello"));               // Output: Invalid input
